Զանգվածները JavaScript-ում տվյալների հիմնարար կառուցվածք են և օգտագործվում են մեկ փոփոխականում բազմաթիվ արժեքներ պահելու համար: Դրանք սահմանվում են քառակուսի փակագծերով [], իսկ արժեքները բաժանվում են ստորակետերով։ Ահա մի օրինակ. 

    
var fruits = ["apple", "banana", "cherry"];

Զանգվածի յուրաքանչյուր արժեք հայտնի է որպես տարր և կարող է մուտք գործել իր ինդեքսով: Ցուցանիշը զրոյի վրա հիմնված ամբողջ թիվ է, որը սկսվում է 0-ից և բարձրանում մինչև n-1, որտեղ n-ը զանգվածի տարրերի թիվն է։ Օրինակ, մրգերի զանգվածի առաջին տարրը հասանելի կլինի այսպես. 

    
fruits[0]; // վերադառնում է "apple"

JavaScript-ի զանգվածները դինամիկ են, և ցանկացած պահի կարող եք տարրեր ավելացնել, հեռացնել կամ փոփոխել: Օրինակ: 

  
fruits.push("orange"); // ավելացնում է տարր զանգվածի վերջում
fruits.pop(); // հեռացնում է զանգվածի վերջին տարրը
fruits.splice(1, 1, "kiwi"); // հեռացնում է 1-ին ինդեքսում գտնվող տարրը և տեղադրում «kivi» այդ դիրքում

Զանգվածներն ունեն նաև մի շարք ներկառուցված մեթոդներ, որոնք թույլ են տալիս տարբեր գործողություններ կատարել տարրերի վրա, ինչպիսիք են տեսակավորումը, զտումը, քարտեզագրումը, կրճատումը և այլն: Ահա մի քանի ընդհանուր օրինակներ. 

  
fruits.sort(); // դասավորում է տարրերը աճման կարգով
fruits.reverse(); // փոխում է տարրերի հերթականությունը
fruits.filter(function(fruit) { return fruit.length > 5; }); // վերադարձնում է զանգված, որի երկարությունը 5-ից մեծ է տարրերով
fruits.map(function(fruit) { return fruit.toUpperCase(); }); // վերադարձնում է մեծատառի վերածված տարրերով զանգված
fruits.reduce(function(total, fruit) { return total + fruit.length; }, 0);// վերադարձնում է տարրերի երկարությունների գումարը

Նկատի ունեցեք, որ JavaScript-ում զանգվածները կարող են պահել տարբեր տիպի տարրեր և նույնիսկ խառնել տարբեր տեսակների: Օրինակ: 

  
var mixed = [1, "apple", true, { name: "John" }];

«activeElement»-ը վերաբերում է վեբ էջի այն տարրին, որը ներկայումս կենտրոնացված է և ընտրված է օգտատերերի փոխազդեցության համար: Սա կարող է լինել մուտքագրման դաշտ, կոճակ, հղում կամ վեբ էջի ցանկացած այլ ինտերակտիվ տարր: Ակտիվ տարրը որոշվում է օգտատիրոջ ներկայիս գտնվելու վայրով, օրինակ՝ մկնիկի սեղմումով, էջի վրայով ներդիրներով կամ ստեղնաշարի նավարկմամբ: Ակտիվ տարրը կարևոր է հասանելիության նպատակներով, քանի որ այն օգնում է օգտատերերին նավարկելու վեբ էջի միջով և շփվել բովանդակության հետ:

«activeElement» հատկությունը փաստաթղթի օբյեկտի մոդելի (DOM) մասն է և հասանելի է JavaScript-ի միջոցով: Այն թույլ է տալիս ծրագրային կերպով մուտք գործել և շահարկել տվյալ պահին ընտրված տարրը վեբ էջի վրա:

Դուք կարող եք օգտագործել «activeElement» հատկությունը՝ իրականացնելու այնպիսի գործողություններ, ինչպիսիք են տարրի կիզակետը փոխելը, դրա ոճը փոխելը կամ այնպիսի իրադարձություններ հրահրել, ինչպիսիք են՝ սեղմելը կամ ուղարկելը: Օրինակ, դուք կարող եք օգտագործել հետևյալ կոդը՝ վեբ էջի վրա կենտրոնացումը որոշակի տարրի վրա փոխելու համար. 

               
  document.activeElement.blur();
  document.getElementById("myInput").focus();

«activeElement» հատկությունը աջակցվում է ժամանակակից վեբ բրաուզերների կողմից, ներառյալ Google Chrome-ը, Mozilla Firefox-ը և Microsoft Edge-ը, ինչպես նաև Internet Explorer 11 և ավելի ուշ տարբերակները:

Ահա մի քանի օրինակ, թե ինչպես կարող եք օգտագործել «activeElement» հատկությունը JavaScript-ում.
